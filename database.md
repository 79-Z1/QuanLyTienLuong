	create or alter proc pr_LuongTheoThang @thang int, @nam int, @manv varchar(10), @HSL float output, @SONC int output, @TU decimal output, @TTC decimal output, @BH decimal output, @THUE decimal output, @LTL decimal output
	as
	begin
		declare @LCS decimal = 1490000
		declare @LTC decimal = 350000
		declare @SNTC int
		declare @LDTT decimal 

		select @HSL = cv.hesoluong
		from CHUCVU cv
		join NHANVIEN nv
		on nv.MaChucVu = cv.MaChucVu
		where nv.MaNV = @manv

		select @TU = pul.SoTien
		from NHANVIEN nv
		join PHIEUUNGLUONG pul
		on pul.MaNV = nv.MaNV
		where nv.MaNV = @manv  and pul.Duyet = 1 and month(pul.NgayUng) = @thang and year(pul.NgayUng) = @nam

		select @SONC = count(cc.MaCong)
		from NHANVIEN nv
		join CHAMCONG cc
		on cc.MaNV = nv.MaNV
		where nv.MaNV = @manv  and month(cc.ngay) = @thang and year(cc.ngay) = @nam and (cc.trangthai = 1 or cc.nghihl = 1)
		group by cc.MaNV
		
		select @SNTC = count(tc.MaNV)
		from NHANVIEN nv
		join TANGCA tc
		on tc.MaNV = nv.MaNV
		where nv.MaNV = @manv  and month(tc.NgayTC) = @thang and year(tc.NgayTC) = @nam
		group by tc.MaNV

		if @SONC is null set @SONC = 0
		if @TU is null set @TU = 0
		if @SNTC is null set @SNTC = 0
		set @BH = @LCS * @HSL * 0.32 * 0.105
		set @LDTT = @LCS * @HSL 
		set @TTC = @SNTC*@LTC
		if @LDTT > 11000000 set @THUE = (@LDTT - 11000000)*0.05
		else set @THUE = 0
		set @LTL = (@LCS * @HSL / 30 * @SONC) + @TTC - @TU - @BH - @THUE
	end
go
